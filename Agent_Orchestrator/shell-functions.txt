# Agent Orchestrator shell functions.
# Install these functions by running node {{HARNESS_ROOT}}/src/install-shell-functions.js eg. `node ~/Repos/Agent_Orchestrator/Agent_Orchestrator/src/install-shell-functions.js`
# Paths are absolute ({{HARNESS_ROOT}} is substituted from the value of "harness-root" in global-config.json at install time) — run from any repo.
hedit()          { code {{HARNESS_ROOT}}/..; }  # hedit — open the harness repo in VS Code

# Topic management
hstartt()        { \node {{HARNESS_ROOT}}/src/start-topic.js "$1" "$2" "$3"; }   # hstartt <topic> [id] [root-repo] — root-repo: abs or relative (resolved vs invocation dir), defaults to cwd; written to topic-config.json as "root-repo"; working/scan root for all agents
hsett()          { \node {{HARNESS_ROOT}}/src/set-topic.js "$1" "$2"; }     # hsett <topic> <id>
hrentopic()      { \node {{HARNESS_ROOT}}/src/rename-topic.js "$1" "$2"; }  # hrentopic <topic|id> <new-name>
hrmtopic()       { \node {{HARNESS_ROOT}}/src/remove-topic.js "$1"; }       # hrmtopic <topic|id|all>

# Unified runner — single command for single-topic and parallel runs.
# Usage: hrun [<id>-]<cmd> [[<id>-]<cmd> ...]
#   hrun 1-c          # topic 1, coding
#   hrun 2-caf        # topic 2, code-assess-fix
#   hrun caf          # last-touched topic, code-assess-fix
#   hrun 1-c 2-caf    # topic 1 coding AND topic 2 code-assess-fix in parallel
# Shorthand: p|c|a|f|af|pc|caf|all|pcaf|cont
hrun()           { \node {{HARNESS_ROOT}}/src/run-parallel.js "$@"; }

# hresume [topic|id|all] — manually trigger auto-resume for topics in the wake queue
# (mostly for testing — this is normally triggered automatically by the OS scheduler at the token-limit reset time).
hresume()        { \node {{HARNESS_ROOT}}/src/auto-resume.js "$@"; }

# Memory (single-file-per-topic — no role argument)
hclear()         { \node {{HARNESS_ROOT}}/src/clear-memory.js "$@"; }    # hclear [topic|id|all]
hscrubmem()      { \node {{HARNESS_ROOT}}/src/scrub-compressed-memory.js "$@"; } # hscrubmem [topic|id|all] — strip stale ## Compressed Memory sections

# Prompt-queue maintenance
# hqregen [topic|id|all] — destructively wipe and re-seed prompt-queue.md for a topic
hqregen()        { \node {{HARNESS_ROOT}}/src/regenerate-queue.js "$@"; }

# Misc
hupdate-models()   { \node {{HARNESS_ROOT}}/src/update-models-reference.js; }  # hupdate-models
hfetch-models()    { \node {{HARNESS_ROOT}}/src/fetch-models.js; }             # hfetch-models — force-refresh model catalog cache
hprobe()           { \node {{HARNESS_ROOT}}/src/run-agent.js --probe; }  # hprobe — checks provider auth
