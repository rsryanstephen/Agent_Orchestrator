# Agent Orchestrator shell functions.

_harness_install_needed() {
    local harness_root="${HARNESS_ROOT:-{{harness_root}}}"
    echo "Please run the install script using \`node $harness_root/src/install-shell-functions.js\`, which will install the shell functions needed." >&2
    return 1
}

# Harness install (idempotent — adds these functions to your rc file if missing and sets your HARNESS_ROOT environment variable). 
# Run with --force to overwrite existing functions.
# node {{harness_root}}/src/install-shell-functions.js "$@"; [--force]
# eg. node ~/Repos/Agent_Orchestrator/Agent_Orchestrator/src/install-shell-functions.js --force

# If these stubs don't run, harness shell functions have not been installed yet.

hedit()          { _harness_install_needed; }  # hedit — open the harness repo in VS Code

# Topic management
hstartt()        { _harness_install_needed; }   # hstartt <topic> [id]
hsett()          { _harness_install_needed; }     # hsett <topic> <id>
hrentopic()      { _harness_install_needed; }  # hrentopic <topic|id> <new-name>
hrmtopic()       { _harness_install_needed; }       # hrmtopic <topic|id|all>

# Unified runner — single command for single-topic and parallel runs.
# Usage: hrun [<id>-]<cmd> [[<id>-]<cmd> ...]
#   hrun 1-c          # topic 1, coding
#   hrun 2-caf        # topic 2, code-assess-fix
#   hrun caf          # last-touched topic, code-assess-fix
#   hrun 1-c 2-caf    # topic 1 coding AND topic 2 code-assess-fix in parallel
# Shorthand: p|c|a|f|af|pc|caf|all|pcaf|cont
hrun()           { _harness_install_needed; }

# hresume [topic|id|all] — manually trigger auto-resume for topics in the wake queue
# (mostly for testing — this is normally triggered automatically by the OS scheduler at the token-limit reset time).
hresume()        { _harness_install_needed; }

# Memory (single-file-per-topic — no role argument)
hclear()         { _harness_install_needed; }    # hclear [topic|id|all]
hscrubmem()      { _harness_install_needed; } # hscrubmem [topic|id|all] — strip stale ## Compressed Memory sections

# Prompt-queue maintenance
# hqregen [topic|id|all] — destructively wipe and re-seed prompt-queue.md for a topic
hqregen()        { _harness_install_needed; }

# Misc
hupdate-models()   { _harness_install_needed; }  # hupdate-models
hfetch-models()    { _harness_install_needed; }             # hfetch-models — force-refresh model catalog cache
hprobe()           { _harness_install_needed; }  # hprobe — checks provider auth
